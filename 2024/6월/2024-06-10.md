1. n 정사각형
```cpp
#include <iostream>
using namespace std;

int main(void)
{
	int input = 0;
	cin >> input;
	for (int i = 0; i < input; i++)
	{
		for (int j = 0; j < input; j++)
			cout << "*";
		cout << '\n';
	}
}			
```

2. 1씩 커지는 삼각형
```cpp
#include <iostream>
using namespace std;

int main(void)
{
	int input = 0;
	cin >> input;
	for (int i = 0; i < input; i++)
	{
		for (int j = input-i-1; j < input; j++)
			cout << '*';
		cout << '\n';
	}
}
```
