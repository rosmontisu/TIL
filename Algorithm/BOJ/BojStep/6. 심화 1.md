#### 5번 boj 1157 단어 공부
```cs
using System;
namespace study_.net
{
    internal class Program2
    {
        static void Main(string[] args)
        {
            //아스키 +32를 하면 소문자가 된다.
            //65(A) + 32 = 97(a)
            //90(Z)

            string word = Console.ReadLine();
            int[] asciiArr = new int[26];

            int wordNum = 0;
            while (wordNum < word.Length)
            {
                for (int i = 0; i < asciiArr.Length; i++)
                {
                    if ((int)word[wordNum] == 65 + i || (int)word[wordNum] == 97 + i)//65이므로 A부터 시작
                    {
                        asciiArr[i]++;
                        break;
                    }
                }
                wordNum++;
            }

            int max = 0;
            int maxAscii = 0;
            int j = 0;

            bool overlap_max = false; // 최대값의 중복을 검사합니다.
            for(; j < asciiArr.Length; j++)
            {
                if (asciiArr[j] == max)
                    overlap_max = true;
                else if(asciiArr[j] > max)
                {
                    max = asciiArr[j];
                    maxAscii = j;
                    overlap_max = false;

                }
            }

            if(overlap_max)
                Console.WriteLine('?');
            else
                Console.WriteLine((char)(maxAscii + 65));
        }
    }
}
```

#### 6번 boj 2941 크로아티아 알파벳
```cpp
#include <iostream>
using namespace std;

int main(void)
{
	string str;
	cin >> str;

	int count = 0;
	for (int i = 0; i < str.length(); i++) {
		if (i == 0) continue;

		if (str[i] == '=') {
			if (str[i - 1] == 'c' || str[i - 1] == 's') count++;
			else if (str[i - 1] == 'z')
			{
				if (i > 1 && str[i - 2] == 'd') count += 2;
				else count++;
			}			
		}
		else if (str[i] == '-') {
			if (str[i - 1] == 'c' || str[i - 1] == 'd') count++;
		}
		else if (str[i] == 'j') {
			if (str[i - 1] == 'l' || str[i - 1] == 'n') count++;
		}
	}

	cout << str.length() - count;
	return 0;
}
```
