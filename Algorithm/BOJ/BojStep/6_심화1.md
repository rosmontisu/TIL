```cpp
#include <iostream>
using namespace std;

int main(void)
{
	string str;
	cin >> str;

	int count = 0;
	for (int i = 0; i < str.length(); i++) {
		if (i == 0) continue;

		if (str[i] == '=') {
			if (str[i - 1] == 'c' || str[i - 1] == 's') count++;
			else if (str[i - 1] == 'z')
			{
				if (i > 1 && str[i - 2] == 'd') count += 2;
				else count++;
			}			
		}
		else if (str[i] == '-') {
			if (str[i - 1] == 'c' || str[i - 1] == 'd') count++;
		}
		else if (str[i] == 'j') {
			if (str[i - 1] == 'l' || str[i - 1] == 'n') count++;
		}
	}

	cout << str.length() - count;
	return 0;
}
```
